 /*                                          

     顯示輸入月份的季節           

 */                                          

                                             

 #include    <iostream.h>                    

                                             

 int main(void)                              

 {  

     int     month;                          

                                             

     cout << "請輸入月份：";         

     cin >> month;                           

                                             

     if (month >= 3 && month <= 5)           

         cout << month << "月是春天。\n";  

     else if (month >= 6 && month <= 8)      

         cout << month << "月是夏天。\n";  

     else if (month >= 9 && month <= 11)     

         cout << month << "月是秋天。\n";  

     else                                    

         cout << month << "月是冬天。\n";  

                                             

     return (0);                             

 }
