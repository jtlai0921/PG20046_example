 //--- ユ传把计x, yぇ ---//

 void swap(int& x, int& y)         

 {                                

     int temp = x;                  

     x = y;                        

     y = temp;                     

 }                                

                                  

 //--- パp欷j逼Cユ传T影鸭 ---//           

 void sort(int& a, int& b, int& c)                        
 {
                                                          

     if (a > b) swap(a, b);                               

     if (b > c) swap(b, c);                               

     if (a > b) swap(a, b);                               

 }
