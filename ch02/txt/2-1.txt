/*

    將輸入的數值除以3（if敘述）

*/
 #include    <iostream.h> 

 int main(void)  

     int     x;   

     cout << "請輸入整數值x：";           

     cin >> x;                                 

     if (x % 3)                                

         cout << "x無法以3整除。\n"; 

     return (0);                               

 }
