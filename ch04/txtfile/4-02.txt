 /*                                        

      輸山九九乘法表                    

 */                                        

                                           

 #include    <iostream.h>                  

                                           

 int main(void)                            

 {                                         

     for (int i = 1; i <= 9; i++) {          

         for (int j = 1; j <= 9; j++)        

             cout << i * j << ' ';          

         cout << '\n';                      

     }                                     

                                           

     return (0);                           

 }
