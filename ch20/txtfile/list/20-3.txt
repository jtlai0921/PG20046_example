 #include    <iostream.h>                

 /*--- qじ兰片n壕慵瓢}CxjMk ---*/     

 int int_search(int x[], int n, int k)                                          

 {

     int found = 0;                                                             

     while (found < n) { 

         if (x[found] == k) break;                                              

         found++;                                                               

     }                                                                          

     return (found);                                                            

 }                                                                              

                                                                                

 /*--- qじ兰片n汗昙瓢}CxjMk ---*/                                 

 int dbl_search(double x[], int n, double k)                                    

 {

     int found = 0;                                                             

     while (found < n) { 

         if (x[found] == k) break;                                              

         found++;                                                               

     }                                                                          

     return (found);                                                            

 }                                                                              

                                                                                

 int main(void)                                                                 

 {

     int x[] = {1, 6, 3, 4, 2, 9, 0, 8};                                        

     int nx = sizeof(x) / sizeof(x[0]);                                         

                                                                                

     int y[] = {1.5, 3.2, 6.8, 4.4, 0.9};                                       

     int ny = sizeof(y) / sizeof(y[0]);                                         

                                                                                

     int pos = int_search(x, nx, 6);                                            

                                                                                

     if (pos == nx)                                                             

         cout << "тぃ6C\n";                                 

     else                                                                       

         cout << "6b材" << pos + 1 << "应旄mC\n";                      

                                                                                

     pos = dbl_search(y, ny, 5.5);                                              

                                                                                

     if (pos == ny)                                                             

         cout << "тぃ5.5C\n";                               

     else                                                                       

         cout << "5.5b材" << pos + 1 << "应旄mC\n";                    

                                                                                

     return (0);                                                                

 }
