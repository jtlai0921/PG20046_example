 /*                                                                              

     單純化的銀行帳戶                                                      

 */                                                                              

                                                                                 

 #include    <iostream.h>                               

                                                       

 int main(void)                                                                  

 {

     long my_account = 0;                          // 我的銀行帳戶裡的存款額        

                                                                                 

     my_account += 100;                            // 存入100元                 

     my_account -=  50;                            // 領出50元                 

     cout << "存款額＝" << my_account << "元\n";   // 顯示存款額                 

                                                                                 

     return (0);                                                                 

 }
