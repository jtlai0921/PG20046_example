/*

    顯示輸入的二個實數之和

*/

                           

 #include    <iostream.h>  

                           

 int main(void)            

 {                         

     double  x;          // 實數x的宣告                                  

     double  y;          //實數y的宣告                                 

                                                                          

     cout << "請輸入x：";    // 提示輸入x    

     cin >> x;               // 將x以整數讀取 

                                                                          

     cout << "請輸入y：";    // 提示輸入y    

     cin >> y;               // 將y以整數讀取    

                                                                          

     cout << "ｘ＋ｙ為" << x + y << "。\n";                           

                                                                          

     return (0);                                                          

 }                                                                        

 執行例子
請輸入x：13.5
請輸入y：27.8
x+y為41.3。
