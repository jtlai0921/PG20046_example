 #include    <iomanip.h>                                                 

 #include    <iostream.h>                                                

                                                                         

 int main(void)                                                          
 {
 
                                                                         

     int pr_no = 0;                          // 借计涵蛹              

     int prime[500];                         // xs借计喊}C       

                                                                         

     for (int x = 2; x <= 1000; x++) {

         for (int i = 2; i < x; i++)                                     

             if ((x % i) == 0)                                           

                 break;                                                  

         if (x == i) prime[pr_no++] = x;     // xO借计

     }                                                                   

                                                                         

     for (int i = 0; i < pr_no; i++)                                     

         cout << setw(4) << prime[i];                                    

     cout << '\n';                                                       

                                                                         

     return (0);                                                         

 }
