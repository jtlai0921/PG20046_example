PetNode* PetList::Bottom(void)                                            

{

    if (top == bottom) return (NULL);                                     

    PetNode* ptr = top;                                                   

    while (ptr->next != bottom)                                           

        ptr = ptr->next;                                                  

    return (ptr);                                                         

}                                                                         

                                                                          

//--- ¶b¶Í¶C∂}¿Y¥°§J§∏Ø¿ ---//                                      

PetList& PetList::Insert(PetNode* x)                                      

{

    PetNode* ptr = top;                                                   

    top = x;                                                              

    top->next = ptr;                                                      

    return (*this);                                                       

}                                                                         

                                                                          

//--- ¶b¶Í¶C•Ω∫›∑sºW§∏Ø¿ ---//                                      

PetList& PetList::Append(PetNode* x)                                      

{ 

    if (top == bottom)                                                    

        Insert(x);                                                        

    else {

        PetNode* ptr = Bottom();                                          

        ptr->next = x;                                                    

        x->next = bottom;                                                 

    }                                                                     

    return (*this);                                                       

}
